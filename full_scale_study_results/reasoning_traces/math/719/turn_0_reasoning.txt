\[
\text{Total weight given away} = \text{Total sacks given away} \times \text{Weight per sack}
\]

\[
\text{Total weight given away} = 7 \text{ sacks} \times 25 \text{ kg/sack}

\[
\text{Total weight given away} = 7 \times 25 = 175 \text{ kg}
