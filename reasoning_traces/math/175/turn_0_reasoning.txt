   \[
   \text{Average Speed} = \frac{1}{3} \times \text{Speed on Flat Sand} + \frac{1}{3} \times \text{Speed on Downhill Slopes} + \frac{1}{3} \times \text{Speed on Uphill Slopes}

   \[
