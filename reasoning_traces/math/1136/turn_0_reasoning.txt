     \[
     2 \times 12 = 24 \text{ marbles}
     \]

   \[
   x + 24 = 60
   \]

   \[
   x + 24 - 24 = 60 - 24
   \]
   \[
   x = 36
   \]

   \[
   \text{Marbles after losing 10} = x - 10 = 36 - 10
   \]
   \[
   = 26
   \]

   \[
   \boxed{26} \text{ marbles}
