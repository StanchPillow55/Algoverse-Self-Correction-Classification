   \[
   \text{Earnings after taxes} = \text{Total earnings} - \text{Tax amount}
   \]
   \[
   \text{Earnings after taxes} = 1200 \, \text{dollars} - 120 \, \text{dollars} = 1080 \, \text{dollars}
   \]
