\[
\text{Time to sauté 90 shrimp} = 90 \text{ shrimp} \times \frac{1}{3} \text{ minutes per shrimp} = 30 \text{ minutes}
\]

\[
\text{Total time} = \text{Time to peel} + \text{Time to sauté} = 15 \text{ minutes} + 30 \text{ minutes} = 45 \text{ minutes}
