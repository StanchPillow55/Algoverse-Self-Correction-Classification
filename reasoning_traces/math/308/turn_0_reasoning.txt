\[
\frac{2 \text{ grams}}{30 \text{ ml}} = \frac{x \text{ grams}}{900 \text{ ml}}
\]

\[
2 \times 900 = 30 \times x
\]

\[
1800 = 30x
\]

\[
x = \frac{1800}{30}
\]
